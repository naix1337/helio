```typescript
// helio-app/backend/src/middleware/auth.ts
import type { Request, Response, NextFunction } from 'express';
import jwt from 'jsonwebtoken';
import type { AuthToken, UserRole } from '../types.js';

export const JWT_SECRET = process.env.HELIO_JWT_SECRET ?? (() => {
  const s = Math.random().toString(36).slice(2) + Math.random().toString(36).slice(2);
  console.warn('[Helio] HELIO_JWT_SECRET not set — sessions will be invalidated on restart. Set it in env for persistence.');
  return s;
})();

declare global {
  namespace Express {
    interface Request {
      user?: AuthToken;
    }
  }
}

export function requireAuth(req: Request, res: Response, next: NextFunction): void {
  const header = req.headers.authorization;
  if (!header?.startsWith('Bearer ')) {
    res.status(401).json({ error: 'Unauthorized' });
    return;
  }
  const token = header.slice(7);
  try {
    const payload = jwt.verify(token, JWT_SECRET) as AuthToken;
    req.user = payload;
    next();
  } catch {
    res.status(401).json({ error: 'Invalid or expired token' });
  }
}

const ROLE_RANK: Record<UserRole, number> = { admin: 3, editor: 2, viewer: 1 };

export function requireRole(minRole: UserRole) {
  return (req: Request, res: Response, next: NextFunction): void => {
    if (!req.user || ROLE_RANK[req.user.role] < ROLE_RANK[minRole]) {
      res.status(403).json({ error: 'Forbidden' });
      return;
    }
    next();
  };
}
```

```typescript
// helio-app/backend/src/routes/auth.ts
import { Router } from 'express';
import bcrypt from 'bcryptjs';
import jwt from 'jsonwebtoken';
import { queries } from '../db/index.js';
import { requireAuth, JWT_SECRET } from '../middleware/auth.js';
import type { UserRole } from '../types.js';

export const authRouter = Router();

const TOKEN_TTL = '7d';

function makeToken(userId: number, role: UserRole): string {
  return jwt.sign({ userId, role }, JWT_SECRET, { expiresIn: TOKEN_TTL });
}

function safeUser(u: { id: number; email: string; name: string; role: string; created_at: number; last_login: number | null }) {
  return { id: u.id, email: u.email, name: u.name, role: u.role, created_at: u.created_at, last_login: u.last_login };
}

// GET /api/auth/setup — is first-run setup needed?
authRouter.get('/setup', (_req, res) => {
  res.json({ needed: queries.countUsers() === 0 });
});

// POST /api/auth/setup — create first admin (only if no users exist)
authRouter.post('/setup', async (req, res) => {
  if (queries.countUsers() > 0) {
    res.status(409).json({ error: 'Setup already completed' });
    return;
  }
  const { email, name, password } = req.body as { email: string; name: string; password: string };
  if (!email || !name || !password || password.length < 8) {
    res.status(400).json({ error: 'email, name, and password (min 8 chars) required' });
    return;
  }
  const hash = await bcrypt.hash(password, 12);
  const id = queries.createUser(email.toLowerCase().trim(), name.trim(), hash, 'admin');
  const user = queries.getUserById(id)!;
  const token = makeToken(id, 'admin');
  res.status(201).json({ token, user: safeUser(user) });
});

// POST /api/auth/login
authRouter.post('/login', async (req, res) => {
  const { email, password } = req.body as { email: string; password: string };
  if (!email || !password) {
    res.status(400).json({ error: 'email and password required' });
    return;
  }
  const row = queries.getUserByEmail(email.toLowerCase().trim()) as
    ({ id: number; email: string; name: string; password_hash: string; role: string; created_at: number; last_login: number | null } | undefined);
  if (!row) {
    res.status(401).json({ error: 'Invalid credentials' });
    return;
  }
  const ok = await bcrypt.compare(password, row.password_hash);
  if (!ok) {
    res.status(401).json({ error: 'Invalid credentials' });
    return;
  }
  queries.updateLastLogin(row.id);
  const token = makeToken(row.id, row.role as UserRole);
  res.json({ token, user: safeUser(row) });
});

// GET /api/auth/me
authRouter.get('/me', requireAuth, (req, res) => {
  const user = queries.getUserById(req.user!.userId);
  if (!user) { res.status(404).json({ error: 'User not found' }); return; }
  res.json(safeUser(user as Parameters<typeof safeUser>[0]));
});
```

```typescript
// helio-app/backend/src/routes/team.ts
import { Router } from 'express';
import bcrypt from 'bcryptjs';
import { queries } from '../db/index.js';
import { requireRole } from '../middleware/auth.js';
import type { UserRole } from '../types.js';

export const teamRouter = Router();

function safeUser(u: { id: number; email: string; name: string; role: string; created_at: number; last_login: number | null }) {
  return { id: u.id, email: u.email, name: u.name, role: u.role, created_at: u.created_at, last_login: u.last_login };
}

// GET /api/team — all users (any authenticated user can view)
teamRouter.get('/', (_req, res) => {
  res.json(queries.getAllUsers().map(safeUser));
});

// POST /api/team — create user (admin only)
teamRouter.post('/', requireRole('admin'), async (req, res) => {
  const { email, name, password, role } = req.body as {
    email: string; name: string; password: string; role: UserRole;
  };
  if (!email || !name || !password || password.length < 8) {
    res.status(400).json({ error: 'email, name, and password (min 8 chars) required' });
    return;
  }
  const validRoles: UserRole[] = ['admin', 'editor', 'viewer'];
  if (!validRoles.includes(role)) {
    res.status(400).json({ error: 'role must be admin, editor, or viewer' });
    return;
  }
  const existing = queries.getUserByEmail(email.toLowerCase().trim());
  if (existing) {
    res.status(409).json({ error: 'Email already in use' });
    return;
  }
  const hash = await bcrypt.hash(password, 12);
  const id = queries.createUser(email.toLowerCase().trim(), name.trim(), hash, role);
  const user = queries.getUserById(id)!;
  res.status(201).json(safeUser(user as Parameters<typeof safeUser>[0]));
});

// PUT /api/team/:id — update role (admin only)
teamRouter.put('/:id', requireRole('admin'), (req, res) => {
  const id = Number(req.params.id);
  const { role } = req.body as { role: UserRole };
  const validRoles: UserRole[] = ['admin', 'editor', 'viewer'];
  if (!validRoles.includes(role)) {
    res.status(400).json({ error: 'role must be admin, editor, or viewer' });
    return;
  }
  queries.updateUserRole(id, role);
  res.json({ ok: true });
});

// DELETE /api/team/:id — delete user (admin only, cannot delete self)
teamRouter.delete('/:id', requireRole('admin'), (req, res) => {
  const id = Number(req.params.id);
  if (id === req.user!.userId) {
    res.status(400).json({ error: 'Cannot delete your own account' });
    return;
  }
  queries.deleteUser(id);
  res.json({ ok: true });
});
```

```typescript
// helio-app/frontend/src/hooks/useAuth.ts
import { useState, useCallback } from 'react';
import type { User } from '../types.ts';

const TOKEN_KEY = 'helio-jwt';

function getStoredToken(): string | null {
  return localStorage.getItem(TOKEN_KEY);
}

function getStoredUser(): User | null {
  try {
    const raw = localStorage.getItem('helio-user');
    return raw ? JSON.parse(raw) : null;
  } catch {
    return null;
  }
}

export function useAuth() {
  const [token, setToken] = useState<string | null>(getStoredToken);
  const [currentUser, setCurrentUser] = useState<User | null>(getStoredUser);

  const login = useCallback((newToken: string, user: User) => {
    localStorage.setItem(TOKEN_KEY, newToken);
    localStorage.setItem('helio-user', JSON.stringify(user));
    setToken(newToken);
    setCurrentUser(user);
  }, []);

  const logout = useCallback(() => {
    localStorage.removeItem(TOKEN_KEY);
    localStorage.removeItem('helio-user');
    setToken(null);
    setCurrentUser(null);
  }, []);

  return { token, currentUser, isAuthenticated: !!token, login, logout };
}

// Singleton token accessor for fetch calls outside React components
export function getAuthToken(): string | null {
  return localStorage.getItem(TOKEN_KEY);
}

// Authenticated fetch wrapper — throws on 401 (caller should redirect to /login)
export async function authFetch(url: string, init?: RequestInit): Promise<Response> {
  const token = getAuthToken();
  const res = await fetch(url, {
    ...init,
    headers: {
      ...init?.headers,
      ...(token ? { Authorization: `Bearer ${token}` } : {}),
    },
  });
  if (res.status === 401) {
    localStorage.removeItem(TOKEN_KEY);
    localStorage.removeItem('helio-user');
    window.location.href = '/login';
  }
  return res;
}
```

```tsx
          fontFamily: 'var(--font-mono)', textTransform: 'uppercase', letterSpacing: '0.06em',
        }}>
          <span>Name</span><span>E-Mail</span><span>Rolle</span><span />
        </div>

        {users.length === 0 && (
          <div style={{ padding: '24px', textAlign: 'center', color: 'var(--text-dim)', fontFamily: 'var(--font-mono)', fontSize: '0.85rem' }}>
            Keine Nutzer gefunden
          </div>
        )}

        {users.map((user, i) => {
          const rc = ROLE_COLORS[user.role as UserRole] ?? ROLE_COLORS.viewer;
          const isSelf = user.id === currentUser?.id;
          return (
            <div key={user.id} style={{
              display: 'grid', gridTemplateColumns: '1fr 1fr auto auto',
              alignItems: 'center', gap: '14px',
              padding: '12px 16px',
              borderBottom: i < users.length - 1 ? '1px solid var(--border)' : 'none',
              fontSize: '0.85rem',
            }}>
              <span style={{ display: 'flex', alignItems: 'center', gap: '10px' }}>
                <div style={{
                  width: '28px', height: '28px', borderRadius: '50%', flexShrink: 0,
                  background: avatarColor(user.email),
                  display: 'flex', alignItems: 'center', justifyContent: 'center',
                  fontSize: '0.65rem', fontWeight: 700, color: '#fff',
                }}>
                  {user.name.slice(0, 2).toUpperCase()}
                </div>
                <span style={{ fontWeight: 500 }}>{user.name}{isSelf && <span style={{ marginLeft: '6px', fontSize: '0.72rem', color: 'var(--text-dim)' }}>(du)</span>}</span>
              </span>
              <span style={{ fontFamily: 'var(--font-mono)', fontSize: '0.8rem', color: 'var(--text-dim)' }}>
                {user.email}
              </span>
              <span>
                {isAdmin ? (
                  <select
                    style={selectStyle}
                    value={user.role}
                    onChange={e => handleRoleChange(user, e.target.value as UserRole)}
                  >
                    <option value="admin">Admin</option>
                    <option value="editor">Editor</option>
                    <option value="viewer">Viewer</option>
                  </select>
                ) : (
                  <span style={{
                    fontSize: '0.75rem', fontFamily: 'var(--font-mono)',
                    padding: '3px 9px', borderRadius: '4px',
                    background: rc.bg, color: rc.color,
                  }}>
                    {user.role}
                  </span>
                )}
              </span>
              <button
                onClick={() => handleDelete(user.id)}
                disabled={!isAdmin || isSelf}
                style={{
                  background: 'none', border: 'none',
                  color: (!isAdmin || isSelf) ? 'var(--surface-3)' : 'var(--text-dim)',
                  cursor: (!isAdmin || isSelf) ? 'not-allowed' : 'pointer',
                  padding: '6px', borderRadius: '6px', display: 'grid', placeItems: 'center',
                }}
              >
                <Trash2 size={15} />
              </button>
            </div>
          );
        })}
      </div>
    </>
  );
}
```
